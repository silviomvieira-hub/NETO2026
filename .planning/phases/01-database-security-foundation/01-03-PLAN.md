```javascript
// js/auth.js — Autenticacao e controle de acesso (AUTH-01 a AUTH-07)
import { supabase } from './supabase.js'

// Cache do perfil do usuario logado
let currentProfile = null

/**
 * Login com email/senha (AUTH-01, per D-06)
 */
export async function login(email, password) {
    const { data, error } = await supabase.auth.signInWithPassword({ email, password })
    if (error) throw error
    currentProfile = null // Reset cache
    return data
}

/**
 * Logout
 */
export async function logout() {
    currentProfile = null
    const { error } = await supabase.auth.signOut()
    if (error) throw error
    window.location.href = '/login.html'
}

/**
 * Restaurar sessao existente (AUTH-02, per D-07)
 * Retorna profile ou null (redireciona para login)
 */
export async function restoreSession() {
    const { data: { session } } = await supabase.auth.getSession()

    if (!session) {
        window.location.href = '/login.html'
        return null
    }

    // Buscar profile com role (per D-08)
    const profile = await getProfile(session.user.id)
    if (!profile) {
        console.error('Profile not found for user', session.user.id)
        await logout()
        return null
    }

    currentProfile = profile
    return profile
}

/**
 * Obter profile do usuario da tabela profiles (per D-08)
 */
async function getProfile(userId) {
    const { data, error } = await supabase
        .from('profiles')
        .select('id, tenant_id, nome, role, telefone, regions, avatar_url')
        .eq('id', userId)
        .single()

    if (error) {
        console.error('Error fetching profile:', error)
        return null
    }

    return data
}

/**
 * Retorna o profile cacheado do usuario atual
 */
export function getCurrentUser() {
    return currentProfile
}

/**
 * Verificar se usuario tem role especifico
 */
export function hasRole(role) {
    return currentProfile?.role === role
}

/**
 * Verificar se usuario e admin (AUTH-03)
 */
export function isAdmin() {
    return currentProfile?.role === 'admin'
}

/**
 * Verificar se usuario e coordenador (AUTH-04)
 */
export function isCoordenador() {
    return currentProfile?.role === 'coordenador'
}

/**
 * Verificar se usuario e cabo eleitoral (AUTH-05)
 */
export function isCaboEleitoral() {
    return currentProfile?.role === 'cabo_eleitoral'
}

/**
 * Trocar senha do usuario logado (AUTH-06, per D-10)
 */
export async function changePassword(newPassword) {
    const { data, error } = await supabase.auth.updateUser({
        password: newPassword
    })
    if (error) throw error
    return data
}

/**
 * Observar mudancas de auth state (AUTH-07, per D-09)
 * Redireciona para login se sessao expirar
 */
export function watchAuthState(callback) {
    supabase.auth.onAuthStateChange((event, session) => {
        if (event === 'SIGNED_OUT') {
            currentProfile = null
            window.location.href = '/login.html'
        }
        if (event === 'TOKEN_REFRESHED' && session) {
            // Sessao renovada — tudo ok
        }
        if (callback) callback(event, session)
    })
}

/**
 * Obter tenant_id do usuario atual
 */
export function getTenantId() {
    return currentProfile?.tenant_id
}

/**
 * Obter regions do usuario atual (para coordenadores)
 */
export function getUserRegions() {
    return currentProfile?.regions || []
}
```
