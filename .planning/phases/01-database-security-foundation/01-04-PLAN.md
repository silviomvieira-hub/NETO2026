```javascript
// js/data.js â€” Supabase CRUD abstraction layer (INFRA-02, per D-17)
import { supabase } from './supabase.js'
import { getCurrentUser, getTenantId } from './auth.js'

// Mapeamento de entidade (nome usado no codigo) para tabela PostgreSQL
const TABLE_MAP = {
    eventos: 'eventos',
    pessoas: 'pessoas',
    lideres: 'lideres',
    quemequem: 'quem_e_quem',
    demandas: 'demandas',
    pesquisas: 'pesquisas',
    materiais: 'materiais',
    gamificacao: 'gamificacao_xp',
    visitas: 'visitas',
    atividades: 'atividades',
    tarefas_game: 'tarefas_game',
    streaks: 'streaks',
    trofeus: 'trofeus'
}

/**
 * Buscar todos os registros de uma entidade (filtrado por RLS automaticamente)
 * @param {string} entity - Nome da entidade (ex: 'eventos', 'pessoas')
 * @param {object} options - { orderBy, ascending, filters }
 * @returns {Promise<Array>}
 */
export async function getData(entity, options = {}) {
    const table = TABLE_MAP[entity]
    if (!table) throw new Error(`Entidade desconhecida: ${entity}`)

    let query = supabase.from(table).select('*')

    // Ordenacao padrao: mais recente primeiro
    const orderBy = options.orderBy || 'created_at'
    const ascending = options.ascending ?? false
    query = query.order(orderBy, { ascending })

    // Filtros opcionais
    if (options.filters) {
        for (const [column, value] of Object.entries(options.filters)) {
            query = query.eq(column, value)
        }
    }

    const { data, error } = await query
    if (error) throw error
    return data || []
}

/**
 * Adicionar um registro a uma entidade
 * Automaticamente adiciona tenant_id e created_by
 * @param {string} entity
 * @param {object} record
 * @returns {Promise<object>} Registro inserido com id
 */
export async function addData(entity, record) {
    const table = TABLE_MAP[entity]
    if (!table) throw new Error(`Entidade desconhecida: ${entity}`)

    const user = getCurrentUser()
    const tenantId = getTenantId()

    // Preparar registro com campos obrigatorios
    const prepared = {
        ...record,
        tenant_id: tenantId,
        created_by: user.id,
        created_at: new Date().toISOString(),
        updated_at: new Date().toISOString()
    }

    // Remover campos que nao devem ser enviados
    delete prepared.id  // Supabase gera UUID automaticamente

    const { data, error } = await supabase
        .from(table)
        .insert(prepared)
        .select()
        .single()

    if (error) throw error
    return data
}

/**
 * Atualizar um registro existente
 * @param {string} entity
 * @param {string} id - UUID do registro
 * @param {object} changes - Campos a atualizar
 * @returns {Promise<object>} Registro atualizado
 */
export async function updateData(entity, id, changes) {
    const table = TABLE_MAP[entity]
    if (!table) throw new Error(`Entidade desconhecida: ${entity}`)

    // Nao permitir alterar campos de sistema
    const sanitized = { ...changes }
    delete sanitized.id
    delete sanitized.tenant_id
    delete sanitized.created_by
    delete sanitized.created_at

    const { data, error } = await supabase
        .from(table)
        .update({
            ...sanitized,
            updated_at: new Date().toISOString()
        })
        .eq('id', id)
        .select()
        .single()

    if (error) throw error
    return data
}

/**
 * Deletar um registro
 * @param {string} entity
 * @param {string} id - UUID do registro
 */
export async function deleteData(entity, id) {
    const table = TABLE_MAP[entity]
    if (!table) throw new Error(`Entidade desconhecida: ${entity}`)

    const { error } = await supabase
        .from(table)
        .delete()
        .eq('id', id)

    if (error) throw error
}

/**
 * Registrar atividade no feed (substitui addAtividade do prototipo)
 * @param {string} texto - Descricao da atividade
 */
export async function logActivity(texto) {
    const user = getCurrentUser()
    await supabase
        .from('atividades')
        .insert({
            tenant_id: getTenantId(),
            user_id: user?.id,
            texto,
            data: new Date().toISOString()
        })
}

/**
 * Exportar todos os dados do tenant como JSON (per D-17)
 * Substitui exportData() do store.js
 */
export async function exportAllData() {
    const entities = Object.keys(TABLE_MAP)
    const exported = {}

    for (const entity of entities) {
        try {
            exported[entity] = await getData(entity, { orderBy: 'created_at', ascending: true })
        } catch (e) {
            console.warn(`Erro ao exportar ${entity}:`, e)
            exported[entity] = []
        }
    }

    // Gerar download
    const blob = new Blob([JSON.stringify(exported, null, 2)], { type: 'application/json' })
    const a = document.createElement('a')
    a.href = URL.createObjectURL(blob)
    a.download = 'campanha-backup-' + new Date().toISOString().split('T')[0] + '.json'
    a.click()
    URL.revokeObjectURL(a.href)

    return exported
}

/**
 * Obter contagem de registros de uma entidade
 */
export async function getCount(entity) {
    const table = TABLE_MAP[entity]
    if (!table) throw new Error(`Entidade desconhecida: ${entity}`)

    const { count, error } = await supabase
        .from(table)
        .select('*', { count: 'exact', head: true })

    if (error) throw error
    return count || 0
}
```

```javascript
// js/migration.js â€” Migracao localStorage -> Supabase (INFRA-02, per D-16/D-17/D-18)
import { supabase } from './supabase.js'
import { getCurrentUser, getTenantId, isAdmin } from './auth.js'
import { STORAGE_KEY, REGIOES_DF } from './constants.js'

/**
 * Verificar se migracao e necessaria (per D-18)
 * Retorna true se localStorage tem dados E Supabase esta vazio
 */
export async function checkMigrationNeeded() {
    const localData = localStorage.getItem(STORAGE_KEY)
    if (!localData) return false

    try {
        const parsed = JSON.parse(localData)
        // Verificar se tem dados reais (nao apenas defaults vazios)
        const hasData = (parsed.eventos?.length > 0) ||
                        (parsed.pessoas?.length > 0) ||
                        (parsed.lideres?.length > 0) ||
                        (parsed.demandas?.length > 0) ||
                        (parsed.gamificacao?.length > 0)
        if (!hasData) return false

        // Verificar se Supabase esta vazio
        const { count } = await supabase
            .from('eventos')
            .select('*', { count: 'exact', head: true })

        const { count: count2 } = await supabase
            .from('pessoas')
            .select('*', { count: 'exact', head: true })

        return (count === 0 && count2 === 0) // Supabase vazio = migracao necessaria
    } catch (e) {
        console.warn('Erro ao verificar migracao:', e)
        return false
    }
}

/**
 * Exportar dados do localStorage como backup antes de migrar
 * Per RESEARCH Pitfall 2: SEMPRE exportar backup antes
 */
export function exportLocalStorageBackup() {
    const raw = localStorage.getItem(STORAGE_KEY)
    if (!raw) throw new Error('Nenhum dado encontrado no localStorage')

    const data = JSON.parse(raw)
    const blob = new Blob([JSON.stringify(data, null, 2)], { type: 'application/json' })
    const url = URL.createObjectURL(blob)
    const a = document.createElement('a')
    a.href = url
    a.download = `campanha-backup-pre-migracao-${new Date().toISOString().split('T')[0]}.json`
    a.click()
    URL.revokeObjectURL(url)

    return data
}

/**
 * Importar dados do localStorage (ou JSON file) para Supabase
 * Per D-16: Mapeia formato existente do exportData() para tabelas Supabase
 *
 * @param {object} localData - JSON com mesma estrutura do localStorage
 * @param {function} onProgress - Callback (entity, count, total) para progress bar
 * @returns {object} Resultados { entity: count, errors: [] }
 */
export async function importToSupabase(localData, onProgress) {
    if (!isAdmin()) throw new Error('Apenas admin pode executar a migracao')

    const user = getCurrentUser()
    const tenantId = getTenantId()
    const results = { migrated: {}, errors: [], total: 0 }

    // Buscar regions para mapear nome -> UUID
    const { data: regions } = await supabase
        .from('regions')
        .select('id, nome')

    const regionMap = {}
    if (regions) {
        regions.forEach(r => { regionMap[r.nome] = r.id })
    }

    // Helper: mapear nome de regiao para UUID
    function getRegionId(nomeRegiao) {
        if (!nomeRegiao) return null
        // Tentar match exato
        if (regionMap[nomeRegiao]) return regionMap[nomeRegiao]
        // Tentar match parcial (caso acentos diferentes)
        const found = Object.keys(regionMap).find(k =>
            k.toLowerCase().includes(nomeRegiao.toLowerCase()) ||
            nomeRegiao.toLowerCase().includes(k.toLowerCase())
        )
        return found ? regionMap[found] : null
    }

    // Helper: inserir batch com tratamento de erro
    async function insertBatch(table, records, entityName) {
        if (!records || records.length === 0) return 0

        try {
            // Inserir em chunks de 50 para evitar timeout
            let inserted = 0
            for (let i = 0; i < records.length; i += 50) {
                const chunk = records.slice(i, i + 50)
                const { error } = await supabase.from(table).insert(chunk)
                if (error) {
                    results.errors.push(`${entityName} (chunk ${Math.floor(i/50)+1}): ${error.message}`)
                } else {
                    inserted += chunk.length
                }
                if (onProgress) onProgress(entityName, inserted, records.length)
            }
            return inserted
        } catch (e) {
            results.errors.push(`${entityName}: ${e.message}`)
            return 0
        }
    }

    // ===== MIGRAR EVENTOS =====
    if (localData.eventos?.length > 0) {
        const mapped = localData.eventos.map(e => ({
            tenant_id: tenantId,
            created_by: user.id,
            titulo: e.titulo || e.nome || 'Sem titulo',
            data: e.data || new Date().toISOString().split('T')[0],
            hora: e.hora || null,
            local: e.local || null,
            tipo: e.tipo || 'Outro',
            observacoes: e.obs || e.observacoes || null,
            region_id: getRegionId(e.regiao),
            regiao_nome: e.regiao || null
        }))
        results.migrated.eventos = await insertBatch('eventos', mapped, 'Eventos')
        results.total += results.migrated.eventos
    }

    // ===== MIGRAR PESSOAS =====
    if (localData.pessoas?.length > 0) {
        const mapped = localData.pessoas.map(p => ({
            tenant_id: tenantId,
            created_by: user.id,
            nome: p.nome || 'Sem nome',
            telefone: p.telefone || null,
            funcao: p.funcao || null,
            regiao_nome: p.regiao || null,
            region_id: getRegionId(p.regiao),
            observacoes: p.obs || p.observacoes || null
        }))
        results.migrated.pessoas = await insertBatch('pessoas', mapped, 'Pessoas')
        results.total += results.migrated.pessoas
    }

    // ===== MIGRAR LIDERES =====
    if (localData.lideres?.length > 0) {
        const mapped = localData.lideres.map(l => ({
            tenant_id: tenantId,
            created_by: user.id,
            nome: l.nome || 'Sem nome',
            telefone: l.telefone || null,
            tipo: l.tipo || null,
            alcance: parseInt(l.alcance) || 0,
            regiao_nome: l.regiao || null,
            region_id: getRegionId(l.regiao),

export async function importFromFile(file, onProgress) {
    return new Promise((resolve, reject) => {
        const reader = new FileReader()
        reader.onload = async (ev) => {
            try {
                const data = JSON.parse(ev.target.result)
                const results = await importToSupabase(data, onProgress)
                resolve(results)
            } catch (err) {
                reject(err)
            }
        }
        reader.onerror = () => reject(new Error('Erro ao ler arquivo'))
        reader.readAsText(file)
    })
}
```
